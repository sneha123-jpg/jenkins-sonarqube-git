pipeline {
    agent any

    stages {

        stage("Pull") {
            steps {
                git 'https://github.com/sneha123-jpg/sonar-jenkin.git'
            }
        }

        stage("Build") {
            steps {
                sh 'mvn clean package'
            }
        }

        stage("Test") {
            steps {

                echo "test success"

                withSonarQubeEnv('my-sonar') {

                    sh '''
                    mvn clean verify org.sonarsource.scanner.maven:sonar-maven-plugin:sonar \
                    -Dsonar.projectKey=java-students \
                    -Dsonar.projectName=java-students
                    '''
                }
            }
        }

        stage("Quality Gate") {
            steps {

                timeout(time: 10, unit: 'MINUTES') {

                    waitForQualityGate abortPipeline: true
                }
            }
        }

        stage("Deploy") {
            steps {
                echo "deploy success"
            }
        }
    }
}
